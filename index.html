2. Correlation

  • What is correlation? : How much of the total variation is explained by co-variation?
a <- 1:100
b <- a + rnorm(100, 0, sqrt(a))
sample <- data.frame(a, b)
plot(sample)



c <- cov(sample)
c[1, 2]/(sqrt(c[1, 1]) * sqrt(c[2, 2]))

## [1] 0.9688

cor(sample)
